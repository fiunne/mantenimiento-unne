} var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action; try { if (action === 'getEdificios') { var sheets = ss.getSheets(); var nombres = sheets.map(s => s.getName()).filter(n => !["MATERIALES", "HISTORIAL", "OT_ABIERTAS"].includes(n)); return crearRespuestaJSON(nombres); } if (action === 'getSectores') { var sheet = ss.getSheetByName(e.parameter.edificio); var datos = sheet.getDataRange().getValues(); return crearRespuestaJSON(datos.slice(1).map(r => ({ sector: r[0], ubi: r[1], inv_aire: r[2], eq_aire: r[3], t_lum: r[4], c_art: r[5], c_lam: r[6], em: r[7], vent: r[8], inv_vent: r[9] }))); } if (action === 'getExtintores') { var ssExt = SpreadsheetApp.openById(ID_EXTINTORES); var sheet = ssExt.getSheetByName(e.parameter.edificio); if (!sheet) return crearRespuestaJSON([]); var datos = sheet.getDataRange().getValues(); return crearRespuestaJSON(datos.slice(1).filter(f => f[3]).map(f => ({ unne: f[3], tipo: f[5], cap: f[6], f_carga: formatFecha(f[7]), f_venc: formatFecha(f[8]), ph: formatFecha(f[10]), planta: f[11], sector: f[12], obs: f[14] }))); } if (action === 'getMateriales') { var sheet = ss.getSheetByName("MATERIALES"); var datos = sheet.getDataRange().getValues(); var obj = {}; datos.slice(1).forEach(r => { if (!obj[r[0]]) obj[r[0]] = {}; if (!obj[r[0]][r[1]]) obj[r[0]][r[1]] = []; obj[r[0]][r[1]].push({nombre: r[2], stock: r[3]}); }); return crearRespuestaJSON(obj); } if (action === 'getAbiertas') { var sheet = ss.getSheetByName("OT_ABIERTAS"); var datos = sheet.getDataRange().getValues(); return crearRespuestaJSON(datos.slice(1).map(r => ({id: r[0], detalle: r[0] + " - " + r[4] + " (" + r[11] + ")" }))); } if (action === 'getHistorial') { var sheet = ss.getSheetByName("HISTORIAL"); var datos = sheet.getDataRange().getValues(); return crearRespuestaJSON(datos.slice(1).reverse().slice(0,15).map(r => ({ id: r[0], fecha: formatFecha(r[1]), edif: r[4], tarea: r[12], tec: r[11], est: r[13] }))); } } catch (err) { return crearRespuestaJSON({error: err.toString()}); } } function doPost(e) { var data = JSON.parse(e.postData.contents); if (data.token !== TOKEN_SEGURIDAD) return ContentService.createTextOutput("No autorizado"); var ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === 'cerrarOT') { var shA = ss.getSheetByName("OT_ABIERTAS"); var shH = ss.getSheetByName("HISTORIAL"); var filas = shA.getDataRange().getValues(); for (var i = 1; i < filas.length; i++) { if (filas[i][0] === data.idOT) { var row = filas[i]; row[13] = "Cerrada"; row.push(data.materiales, data.comentario, new Date()); shH.appendRow(row); shA.deleteRow(i + 1); return ContentService.createTextOutput("Ok"); } } } if (data.action === 'updateRelevamiento') { var sheet = ss.getSheetByName(data.edificio); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][1] === data.ubi && rows[i][0] === data.sector) { sheet.getRange(i + 1, 3, 1, 8).setValues([[data.inv_aire, data.eq_aire, data.t_lum, data.c_art, data.c_lam, data.em, data.vent, data.inv_vent]]); return ContentService.createTextOutput("Ok"); } } } // Nueva OT: no se envía "action" if (!data.action) { var sheet = ss.getSheetByName("OT_ABIERTAS"); var id = "OT-" + Math.floor(Math.random() * 9000 + 1000); sheet.appendRow([id, new Date(), data.tipo, data.especialidad, data.edificio, data.sector, data.ubicacion, data.nInventario, data.equipo, data.reportadoPor, data.medioReporte, data.personal, data.descripcion, "Abierta"]); return ContentService.createTextOutput("Ok"); } } function formatFecha(f) { if (!f || !(f instanceof Date)) return f; return Utilities.formatDate(f, "GMT-3", "dd/MM/yyyy"); } function crearRespuestaJSON(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
